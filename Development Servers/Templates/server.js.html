How to Set Up a Secure Node.js Server

Step 1: Install Node.js

Make sure you have Node.js installed on your system. You can download it from the official Node.js website.

Step 2: Create a Project Directory

mkdir my-node-server
cd my-node-server

Step 3: Initialize a Node.js Project

Run the following command to create a package.json file:

npm init -y

Step 4: Install Dependencies

Install the necessary packages:

npm install express dotenv cors csurf helmet cookie-parser

Step 5: Create a .env File

Create a file named .env in your project root:

touch .env

Add any environment variables here, like the port number:

PORT=3000

Step 6: Create the Server File

Create a file named server.js:

touch server.js

Open server.js in your favorite text editor and add the following code:

require('dotenv').config();
const express = require('express');
const https = require('https');
const fs = require('fs');
const cors = require('cors');
const csurf = require('csurf');
const helmet = require('helmet');
const cookieParser = require('cookie-parser');

const app = express();

// Middleware for CORS
app.use(cors({
    origin: 'https://your-frontend-domain.com', // Change this to your frontend domain
    credentials: true // Allow cookies to be sent
}));

// Middleware for security headers
app.use(helmet());

// Middleware to parse cookies
app.use(cookieParser());

// Middleware to parse JSON bodies
app.use(express.json());

// CSRF protection
const csrfProtection = csurf({
    cookie: true
});
app.use(csrfProtection);

// Middleware to generate session tokens
app.use((req, res, next) => {
    if (!req.session) {
        req.session = {}; // Initialize session if it doesn't exist
    }
    req.session.token = generateSessionToken();
    next();
});

// Generate a random session token
function generateSessionToken() {
    return Math.random().toString(36).substr(2);
}

// Example route with CSRF protection
app.get('/csrf-token', (req, res) => {
    res.json({ csrfToken: req.csrfToken() });
});

// Example route
app.post('/api/data', (req, res) => {
    const { inputData } = req.body;
    res.json({ message: 'Data received successfully' });
});

// Start HTTPS server
const sslOptions = {
    key: fs.readFileSync('ssl/key.pem'),
    cert: fs.readFileSync('ssl/cert.pem')
};

const PORT = process.env.PORT || 3000;
https.createServer(sslOptions, app).listen(PORT, () => {
    console.log(\`Server is running securely on https://localhost:\${PORT}\`);
});

Step 7: Start the Server

Run the server with the following command:

node server.js

Open your browser and navigate to https://localhost:3000 to see your server running!
